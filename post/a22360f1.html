def traversingFolder(path, crack_file):
    files = os.listdir(path)
    for file in files:
        file = os.path.join(path, file)
        if os.path.isdir(file):
            if not os.listdir(file):
                continue
            else:
                traversingFolder(file, crack_file)
        else:
            doCheck(file, crack_file)

        if x == 0:
